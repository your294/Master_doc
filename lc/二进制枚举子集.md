1681 最小不兼容性
先预处理出所有符合条件的子集，然后二进制遍历子集，利用dp递推即可，最后答案是用完所有数的dp位置
```cpp
class Solution {
public:
    int minimumIncompatibility(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(1 << n, INT_MAX);
        dp[0] = 0;
        int group = n / k;
        unordered_map<int, int> values;
        // preprocess the sub set for satisfied the requirements
        for (int mask = 1; mask < (1 << n); ++mask) {
            if (__builtin_popcount(mask) != group) continue;
            int mn = 20, mx = 0;
            unordered_set<int> cur;
            for (int i = 0; i < n; ++i) {
                if (mask & (1 << i)) {
                    if (cur.count(nums[i]) > 0) {
                        break;
                    }
                    cur.insert(nums[i]);
                    mn = min(mn, nums[i]);
                    mx = max(mx, nums[i]);
                }
            }
            if (cur.size() == group) {
                values[mask] = mx - mn;
            }
        }
        // dp and state zip
        for (int mask = 0; mask < (1 << n); ++mask) {
            if (dp[mask] == INT_MAX) continue;
            unordered_map<int, int> seen;
            for (int i = 0; i < n; ++i) {
                if ((mask & (1 << i)) == 0) {
                    seen[nums[i]] = i;
                }
            }
            if (seen.size() < group) {
                continue;
            }
            int sub = 0;
            for (auto& pair : seen) {
                sub |= (1 << pair.second);
            }
            // 降序遍历sub的非空子集
            int nxt = sub;
            while (nxt > 0) {
                if (values.count(nxt) > 0) {
                    // dp[mask]以mask的dp作为集合, 再加上以nxt为集合的不兼容性
                    dp[mask | nxt] = min(dp[mask | nxt], dp[mask] + values[nxt]);
                }
                nxt = (nxt - 1) & sub;
            }
        }
        return (dp[(1 << n) - 1] < INT_MAX) ? dp[(1 << n) - 1] : -1;
    }
};
```
